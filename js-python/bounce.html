import turtle

G = 9.81 # m/s^2
friction = 0.04

def move(ball):
    x, y = ball.position()
    dx, dy = ball.velocity
    
    dy = dy - G
    x = x + dx
    y = y + dy
    
    if not (-200 < x < 200):
        dx = -dx
        x = x + dx
    if not (-200 < y < 200):
        dy = -dy
        y = y + dy
    
    dx = dx * (1-friction)
    dy = dy * (1-friction)
        
    ball.velocity = (dx, dy)
    ball.setposition(x, y)


ball1 = turtle.Turtle()
ball1.speed(5)
ball1.shape("circle")
ball1.color("red")
ball1.penup()
ball1.velocity = (20, 50)

for i in range(500):
    move(ball1)
